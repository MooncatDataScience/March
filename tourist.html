Avg_city <- function(year){
  plt <- rdf %>%
    select(Total, Year, City) %>%
    filter(Year == 2014) %>%
    group_by(City) %>%
    summarize(mean_city = mean(Total)) %>%
    arrange(desc(mean_city)) %>%
    top_n(10)
  
  print(plt)
  
  ggplot(data = plt, aes(x = City, y = mean_city, fill = City)) +
  geom_bar(stat="identity") +
  theme_bw()
}
